export function baseHeaderEq(
    a: { block: CalloutBlock; collapsed: boolean },
    b: { block: CalloutBlock; collapsed: boolean }
): boolean {
    return (
        a.block.type === b.block.type &&
        a.block.title === b.block.title &&
        a.collapsed === b.collapsed &&
        a.block.headerFrom === b.block.headerFrom &&
        a.block.hasContent === b.block.hasContent
    );
}

    });
